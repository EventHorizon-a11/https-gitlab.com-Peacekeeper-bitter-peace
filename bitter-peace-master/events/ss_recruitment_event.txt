# SS-Division 'LSSAH' - Germany
country_event = {
	id = BP_ss_recruitment_event.1
	title = BP_ss_recruitment_event.1.t	
	desc = BP_ss_recruitment_event.1.d	
	picture = GFX_report_event_ss_lssah	 

	is_triggered_only = yes

	option = {
		name = BP_ss_recruitment_event.1.a

			limit = {
				NOT = { has_country_flag = SS_motorised_regiment }
			}
			SS_motorised_regiment = yes
		}
		set_country_flag = SS_lssah_regiment
		custom_effect_tooltip = GER_regiment_lssah_tt

			random_owned_controlled_state = {
				limit = { ROOT = { has_full_control_of_state = PREV } }
				prioritize = { 64 }
				create_unit = {
				division = "name = \"Infanterie-Regiment 'LSSAH' (mot.)\" division_template = \"SS-Regiment\" start_experience_factor = 0.2 start_equipment_factor = 1" 
					owner = GER
				}
			}
		}
	}
}
